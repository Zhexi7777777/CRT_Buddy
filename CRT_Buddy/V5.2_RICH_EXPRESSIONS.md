```python
def draw_mascot(self, painter):
    if self.current_mood == "happy":
        # 星星眼
        画星星()
        # 大笑
        画曲线()
        # 绿色光晕
        设置颜色(绿色)

        # 旋转眼

        画旋转动画()
        # O形嘴
```
